w = 1.0  # peso (parametro) inicial, al azar

# modelo: calculo de la salida ("forward pass")
def forward(x):
    return x * w

# funcion de activacion
def error(x, y):
    y_pred = forward(x)
    return (y_pred - y) * (y_pred - y)

# calculo del gradiente de la activacion con respecto al peso
def gradient(x, y):  # d_loss/d_w
    return 2 * x * (x * w - y)
